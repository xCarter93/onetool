    ```typescript
    export const listNamesForOrg = query({
      args: {},
      handler: async (ctx) => {
        const orgId = await getOptionalOrgId(ctx);
        if (!orgId) return [];

        const clients = await ctx.db
          .query("clients")
          .withIndex("by_org", (q) => q.eq("orgId", orgId))
          .collect();

        return clients.map((c) => ({
          _id: c._id,
          companyName: c.companyName,
        }));
      },
    });
    ```

    Use `getOptionalOrgId` (not `getCurrentUserOrgId`) to match the existing `list` query pattern — returns empty array instead of throwing when no auth context.
